    • Write a function that returns the factorial of a number. 
      def factorial(n):
    if n == 0 or n == 1:
        return 1
    return n * factorial(n - 1)
    • Create a function that checks if a string is a palindrome. 
      def is_palindrome(text):
    text = text.lower()              # normalize case
    text = text.replace(" ", "")     # optional: remove spaces
    return text == text[::-1]        # compare with reversed string

# Example
word = input("Enter a word: ")
if is_palindrome(word):
    print("It is a palindrome.")
else:
    print("It is not a palindrome.")

    • Simple calculator (add, subtract, multiply, divide). 
      def calculator(a, b, operation):
    if operation == "+":
        return a + b
    elif operation == "-":
        return a - b
    elif operation == "*":
        return a * b
    elif operation == "/":
        return a / b
    else:
        return "Invalid operation"

# Example
num1 = float(input("Enter first number: "))
op = input("Enter operation (+, -, *, /): ")
num2 = float(input("Enter second number: "))

print("Result:", calculator(num1, num2, op))

    • To-do list program with add/remove/display options. 
      todo = []

while True:
    print("\n--- TO-DO LIST MENU ---")
    print("1. Add a task")
    print("2. Remove a task")
    print("3. Display tasks")
    print("4. Exit")

    choice = input("Choose an option: ")

    if choice == "1":
        task = input("Enter a new task: ")
        todo.append(task)
        print("Task added.")

    elif choice == "2":
        task = input("Enter the task to remove: ")
        if task in todo:
            todo.remove(task)
            print("Task removed.")
        else:
            print("Task not found.")

    elif choice == "3":
        print("\nYour To-Do List:")
        if not todo:
            print("No tasks yet.")
        else:
            for i, t in enumerate(todo, start=1):
                print(i, "-", t)

    elif choice == "4":
        print("Goodbye!")
        break

    else:
        print("Invalid option. Try again.")
